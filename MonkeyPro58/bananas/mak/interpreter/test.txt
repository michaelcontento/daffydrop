
Class C

	Field x=1
	
	Method New()
		Print "New C!"
	End
	
	Method Update()
		Print "C.Update!"
	End
End

Class D Extends C

	Method New()
		x=2
	End
End

Function Main()

	Local list:=New List<C>
	
	list.AddLast New C
	list.AddLast New D
	list.AddLast New C
	list.AddLast New D

	For Local c:=Eachin list
		Print c.x
	Next
	
	Print "Bye!"	
	
End
